استعراض الملفات والمجلدات

التنقل وعرض محتوى النظام

ملفات عرض الملفات في المجلد
يعرض قائمة بجميع الملفات والمجلدات في المسار الحالي
bash
$ls
$ls -l # تفاصيل + الصلاحيات
$ls -la # يشمل الملفات المخفية
$ls -lh # حجم الملف بشكل مقروء
$ls -lt # ترتيب حسب التاريخ
$ls -ltr # ترتيب عكسي حسب التاريخ
$ls /etc # عرض مجلد معين
ملفات عرض المسار الحالي
يظهر المسار الكامل للمجلد الذي أنت فيه الآن
bash
$pwd
#→ /home/ayoub/projects
$pwd -P # المسار الفعلي (بدون روابط رمزية)
نظام معرفة المستخدم الحالي
يعرض اسم المستخدم الذي تعمل بصلاحياته الآن
bash
$whoami
$id # يعرض ID + المجموعات
$who # المستخدمون المتصلون
$w # تفاصيل المتصلين
$last # آخر عمليات تسجيل الدخول
ملفات التنقل بين المجلدات
الانتقال من مجلد لآخر في نظام الملفات
bash
$cd /var/www # الانتقال لمسار معين
$cd .. # المجلد السابق
$cd ~ # المجلد الرئيسي (Home)
$cd - # المجلد السابق
$cd ../.. # مستويين للأعلى
